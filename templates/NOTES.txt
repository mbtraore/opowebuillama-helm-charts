Your Open WebUI instance can be accessed through the following DNS name from within your cluster:

    {{ include "webui.fullname" . }}.{{ .Values.ingress.hostname  }} (port {{ .Values.service.servicePortHttps }})

To access your Open WebUI instance from outside the cluster follow the steps below:

{{- if .Values.ingress.enabled }}

1. Get Open WebUI URL and associated Hostname to your cluster external IP:

   export CLUSTER_IP=$(minikube ip) # On Minikube. Use: `kubectl cluster-info` on others K8s clusters
   echo "Open WebUI URL: http{{ if .Values.ingress.tls.enabled }}s{{ end }}://{{ .Values.ingress.hostname }}/"
   echo "$CLUSTER_IP  {{ .Values.ingress.hostname }}" | sudo tee -a /etc/hosts

{{- else }}
{{- $port := .Values.service.servicePortHttp | toString }}

1. Get Open WebUI URL by running these commands:

{{- if contains "NodePort" .Values.service.type }}

   export NODE_PORT=$(kubectl get --namespace {{ .Values.ingress.hostname  }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "webui.fullname" . }})
   export NODE_IP=$(kubectl get nodes --namespace {{ .Values.ingress.hostname  }} -o jsonpath="{.items[0].status.addresses[0].address}")
   echo "Open WebUI URL: http://$NODE_IP:$NODE_PORT/"

{{- else if contains "LoadBalancer" .Values.service.type }}

  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        Watch the status with: 'kubectl get svc --namespace {{ .Values.ingress.hostname  }} -w {{ include "webui.fullname" . }}'

   export SERVICE_IP=$(kubectl get svc --namespace {{ .Values.ingress.hostname  }} {{ include "webui.fullname" . }} --template "{{ "{{ range (index .status.loadBalancer.ingress 0) }}{{ . }}{{ end }}" }}")
   echo "Open WebUI URL: http://$SERVICE_IP{{- if ne $port "80" }}:{{ .Values.service.servicePortHttp }}{{ end }}/"

{{- else if contains "ClusterIP"  .Values.service.type }}

   kubectl port-forward --namespace {{ .Values.ingress.hostname  }} svc/{{ include "webui.fullname" . }} {{ .Values.service.servicePortHttp }}:{{ .Values.service.servicePortHttp }} &
   echo "Open WebUI URL: http://127.0.0.1{{- if ne $port "80" }}:{{ .Values.service.servicePortHttp }}{{ end }}//"

{{- end }}
{{- end }}

2. Open a browser and access Open WebUI using the obtained URL.
